    <script>
        function openTab(event, tabId) {
            // Sayfanın yenilenmesini veya yukarı zıplamasını engelle
            if(event) event.preventDefault();

            // 1. Tüm butonları pasif yap
            document.querySelectorAll('.sub-menu-btn').forEach(btn => btn.classList.remove('active'));
            document.querySelectorAll('.dropdown-item').forEach(item => item.classList.remove('active'));

            // 2. Tıklanan butonu aktif yap
            if(event) {
                event.currentTarget.classList.add('active');
            } else {
                // Eğer event yoksa (sayfa ilk yüklendiğinde URL'den geliyorsa)
                let activeBtn = document.querySelector(`[onclick*="${tabId}"]`);
                if(activeBtn) activeBtn.classList.add('active');
            }

            // 3. Tüm içerik alanlarını gizle
            document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active'));

            // 4. Sadece seçilen 'tabId'ye sahip içeriği göster
            let targetContent = document.getElementById(tabId);
            if(targetContent) targetContent.classList.add('active');

            // 5. URL'yi güncelle (sayfa yenilenmeden geçmişe ekler)
            if(history.pushState) {
                history.pushState(null, null, '#' + tabId);
            }

            const dropdown = document.querySelector('.sub-menu-dropdown');
            if (dropdown) {
                dropdown.removeAttribute('open');
            }
        }

        // Sayfa ilk yüklendiğinde URL'de #varsa (örn: .../Kurumsal/Index#misyon) o sekmeyi aç
        document.addEventListener('DOMContentLoaded', function() {
            let hash = window.location.hash.substring(1);
            if(hash && document.getElementById(hash)) {
                openTab(null, hash);
            }
        });
    </script>
}
